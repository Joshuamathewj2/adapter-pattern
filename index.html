# adapters.py — The heart of the pattern

from weather_provider import WeatherProvider
from services import OpenWeatherMapAPI, WttrInAPI, WeatherXMLService

class OpenWeatherAdapter(WeatherProvider):
    """Adapts JSON/Kelvin API → WeatherProvider interface"""
    def __init__(self):
        self._api = OpenWeatherMapAPI()   # Wraps the incompatible service

    def get_temperature(self, city):
        data = self._api.fetch_data(city)
        kelvin = data["main"]["temp"]
        return kelvin - 273.15             # ADAPTATION: Kelvin → Celsius

class WttrInAdapter(WeatherProvider):
    """Adapts plain text API → WeatherProvider interface"""
    def __init__(self):
        self._api = WttrInAPI()

    def get_temperature(self, city):
        raw = self._api.get_weather_string(city)
        # "Weather in Chennai: +35C, ..." → 35.0
        return float(raw.split("+")[1].split("C")[0])

class XMLWeatherAdapter(WeatherProvider):
    """Adapts XML string API → WeatherProvider interface"""
    def __init__(self):
        self._api = WeatherXMLService()

# weather_provider.py — The Target Interface
# All adapters MUST implement this contract

from abc import ABC, abstractmethod

class WeatherProvider(ABC):

    @abstractmethod
    def get_temperature(self, city: str) -> float:
        """Return temperature in Celsius."""
        pass

    @abstractmethod
    def get_description(self, city: str) -> str:
        """Return a short weather description."""
        pass

    @abstractmethod
    def get_source_name(self) -> str:
        """Return the name of the data source."""
        pass
# services.py — Incompatible legacy/3rd-party services
# We CANNOT change these classes!

class OpenWeatherMapAPI:
    """Returns nested dict with temp in KELVIN"""
    def fetch_data(self, city):
        return {"main": {"temp": 308.15},
                "weather": [{"description": "hot and humid"}]}

class WttrInAPI:
    """Returns plain TEXT string"""
    def get_weather_string(self, city):
        return f"Weather in {city}: +35C, Condition: hot"

class WeatherXMLService:
    """Returns raw XML string (legacy!)"""
    def get_xml_data(self, city):
        return f"<weather><temp_celsius>35</temp_celsius></weather>"
# main.py — Client code
# Uses ONLY the WeatherProvider interface.
# Has NO idea what format each adapter handles!

from adapters import OpenWeatherAdapter, WttrInAdapter, XMLWeatherAdapter

def display_weather(provider: WeatherProvider, city: str):
    temp  = provider.get_temperature(city)
    desc  = provider.get_description(city)
    print(f"  🌡  {city}: {temp}°C — {desc}")

providers = [
    OpenWeatherAdapter(),   # JSON / Kelvin
    WttrInAdapter(),         # Plain text
    XMLWeatherAdapter(),     # Legacy XML
]

for provider in providers:
    # Same function call — different source every time!
    display_weather(provider, "Chennai")
Key Insight for Viva: The client (main.py) calls the exact same method on all 3 providers.
Behind the scenes, each adapter speaks a different language — JSON, plain text, XML.
The client never knows. That's the entire point of the Adapter Pattern.
